115     /**
116      * Maakt de parameter string, onafhankelijk van de toegangsmethode (POST of
117      * GET).
118      * 
119      * @param request
120      *            het request van dit service verzoek
121      * @param excludeparam
122      *            een request parameter die eventueel uitgesloten dient te
123      *            worden
124      * @return de string begint met een ampersand (&amp;)
125      */
126     protected final String buildQueryString(HttpServletRequest request,
127             String excludeparam) {
128         /* want servlet api is niet generic */
129         @SuppressWarnings("unchecked")
130         final Map<String, String[]> reqs = request.getParameterMap();
131         final StringBuilder s = new StringBuilder();
132         String key;
133         String[] vals;
134 
135         for (final Map.Entry<String, String[]> entry : reqs.entrySet()) {
136             key = entry.getKey();
137             if (key.equals(excludeparam)) {
138                 continue;
139             }
140             s.append("&amp;").append(key).append("=");
141             vals = entry.getValue();
142             final int count = vals.length;
143             if (count > 1) {
144                 for (int i = 0; i < count; i++) {
145                     s.append(vals[i]);
146                     s.append("&amp;").append(key).append("=");
147                 }
148                 s.delete(s.lastIndexOf("&amp;"), s.length());
149             } else {
150                 s.append(vals[0]);
151             }
152         }
153         // dit moet soms wel soms niet..dus maar even niet: s.delete(0,
154         // "&amp;".length());
155         return s.toString();
156     }

158     /**
159      * Geeft een paginering in HTML formaat. Als in een resultaatlijst
160      * (bijvoorbeeld bij de zoekpagina) meer dan een configureerbaar aantal
161      * resultaten gevonden worden, komen er meerdere pagina's met
162      * zoekresultaten. In dit geval komt er boven en onder de resultaatlijst een
163      * paginanummering te staan. Deze bestaat uit een aantal elementen:
164      * <dl>
165      * <dt>Vorige</dt>
166      * <dd>Indien de huidige resultaatpagina niet de eerste is begint de
167      * paginanummering met een link 'Vorige'</dd>
168      * <dt>Volgende</dt>
169      * <dd>Indien de huidige resultaatpagina niet de laatste is eindigt de
170      * paginanummering met een link 'Volgende'</dd>
171      * <dt>Paginanummers</dt>
172      * <dd>Een maximum van 7 paginanummers staat in de lijst. Dit is de huidige
173      * pagina, twee ervoor, twee erna en de eerste en laatste pagina. De
174      * volgende regels gelden hiervoor:
175      * <ul>
176      * <li>Indien er geen twee pagina's voor de huidige pagina zijn worden er
177      * meer nummers erna getoond. En vice versa. Het totaal van deze rij blijft
178      * altijd 5.</li>
179      * <li>Indien in de twee pagina's voor of na de huidige pagina (of in de
180      * uitzonderingsregel hierboven) de eerste of laatste pagina is opgenomen
181      * vervalt deze als aparte optie.</li>
182      * <li>Tussen de rij van 5 paginanummers en de eerste danwel laatste is een
183      * extra item met '...'.</li>
184      * <li>Afgezien van 'Vorige', 'Volgende' en '...' worden er alleen numerieke
185      * waarden getoond (de paginanummers)</li>
186      * </ul>
187      * </dd>
188      * </dl>
189      * 
190      * @param items
191      *            aantal items
192      * @param request
193      *            the request
194      * @return een html snippet met hyperlinks of een lege string als er geen
195      *         paginering is
196      */
197     protected final String buildPageList(int items, HttpServletRequest request) {
198         final int totaalAantalPaginas = (int) Math.ceil((double) items
199                 / this.itemsPerPage);
200         LOGGER.debug("Er komen " + totaalAantalPaginas + " pagina's");
201         if (totaalAantalPaginas <= 1) {
202             // bij 1 (of minder) geen paginering, klaar!
203             return "";
204         }
205 
206         int currentOffset;
207         try {
208             currentOffset = Integer.parseInt(request
209                     .getParameter(REQ_PARAM_PAGEOFFSET.code));
210         } catch (final NullPointerException e) {
211             currentOffset = 0;
212         } catch (final NumberFormatException e) {
213             currentOffset = 0;
214         }
215         final String reqString = this.buildQueryString(request,
216                 REQ_PARAM_PAGEOFFSET.code);
217 
218         final String addParam = request.getServletPath().replaceFirst("/", "")
219                 + "?" + reqString + "&amp;" + REQ_PARAM_PAGEOFFSET + "=";
220 
221         final int eerste = 0 * this.itemsPerPage;
222         final int vorige = currentOffset - this.itemsPerPage;
223         final int volgende = currentOffset + this.itemsPerPage;
224         final int laatste = (totaalAantalPaginas - 1) * this.itemsPerPage;
225 
226         final StringBuilder sb = new StringBuilder(
227                 "<ul class=\"geozetPaging\">");
228         // vorige
229         if (vorige >= 0) {
230             sb.append("<li class=\"prev first\"><a href=\"").append(addParam)
231                     .append(vorige).append("\">")
232                     .append(this._RESOURCES.getString("KEY_PAGINERING_VORIGE"))
233                     .append("</a></li>");
234         }
235         // eerste pag
236         if (eerste != currentOffset) {
237             sb.append(this.maakPaginaItem(addParam, eerste));
238         }
239 
240         LOGGER.debug("eerste=" + eerste + " huidige=" + currentOffset);
241         // ... tussendoor
242         if (totaalAantalPaginas > 6) {
243             // alleen als aantal pagina's groter is dan 5
244             if ((currentOffset - eerste) > (3 * this.itemsPerPage)) {
245                 // alleen als er meer dan 3 paginas tussen eerste en huidige
246                 // zitten
247                 sb.append("<li>...</li>");
248             }
249         }
250 
251         // bepalen welke pagina's voorafgaand en volgend op
252         int start = currentOffset - (2 * this.itemsPerPage);
253         int maxPages = 5;
254         if (start < 0) {
255             // minimaal bij offset 0 beginnen
256             start = 0;
257         }
258 
259         if (currentOffset >= ((totaalAantalPaginas - 3) * this.itemsPerPage)) {
260             // maximaal bij totaal aantal minus 5 eindigen
261             start = (totaalAantalPaginas * this.itemsPerPage)
262                     - (maxPages * this.itemsPerPage);
263             if (start < 0) {
264                 // voor het geval er minder dan 5 pagina's zijn
265                 start = 0;
266                 maxPages = totaalAantalPaginas;
267             }
268 
269         }
270         // pagina's
271         int offset = 0;
272         for (int i = 0; i < maxPages; i++) {
273             offset = start + (i * this.itemsPerPage);
274             if (currentOffset == offset) {
275                 LOGGER.debug("pagina " + (i + 1)
276                         + " is de huidige pagina en heeft offset " + offset);
277                 sb.append("<li class=\"active\"><strong>")
278                         .append((offset / this.itemsPerPage) + 1)
279                         .append("</strong></li>");
280             } else {
281                 if (!((eerste == offset) || (laatste == offset))) {
282                     // eerste en laatste overslaan, die komen er sowieso in
283                     LOGGER.debug("pagina " + (i + 1) + " heeft offset "
284                             + offset);
285                     sb.append(this.maakPaginaItem(addParam, offset));
286                 }
287             }
288         }
289 
290         // ... tussendoor
291         if (totaalAantalPaginas > 6) {
292             if ((laatste - currentOffset) > (3 * this.itemsPerPage)) {
293                 // alleen als er meer dan 3 paginas tussen laatste en huidige
294                 // zitten
295                 sb.append("<li>...</li>");
296             }
297         }
298 
299         // laatste pag
300         if (laatste != currentOffset) {
301             sb.append(this.maakPaginaItem(addParam, laatste));
302         }
303         // volgende alleen als we nog niet aan het einde zijn
304         if (volgende <= laatste) {
305             sb.append("<li class=\"next\"><a href=\"")
306                     .append(addParam)
307                     .append(volgende)
308                     .append("\">")
309                     .append(this._RESOURCES
310                             .getString("KEY_PAGINERING_VOLGENDE"))
311                     .append("</a></li>");
312         }
313         sb.append("</ul>");
314         return sb.toString();
315     }

319      * 
320      * @param addParam
321      *            request string snipper
322      * @param plusPag
323      *            pagina offset snipper
324      * @return pagina hyperlink

326     private String maakPaginaItem(String addParam, int plusPag) {

433     protected double[] parseLocation(HttpServletRequest request)
434             throws ServletException {
435         try {
436             // request params uitlezen voor het zoeken
437             final double xcoord = Double.valueOf(request
438                     .getParameter(REQ_PARAM_XCOORD.code));
439             final double ycoord = Double.valueOf(request
440                     .getParameter(REQ_PARAM_YCOORD.code));
441             final double straal = Double
442                     .valueOf((null == request
443                             .getParameter(REQ_PARAM_STRAAL.code) ? OPENLS_ZOOMSCALE_STANDAARD
444                             .toString() : request
445                             .getParameter(REQ_PARAM_STRAAL.code)));
446             LOGGER.debug("request params:" + xcoord + ":" + ycoord + " straal:"
447                     + straal);
448             return new double[] { xcoord, ycoord, straal };
449         } catch (final NullPointerException e) {
450             LOGGER.error(
451                     "Een van de vereiste parameters werd niet in het request gevonden.",
452                     e);
453             throw new ServletException(e);
454         } catch (final NumberFormatException e) {
455             LOGGER.error(
456                     "Een van de vereiste parameters kon niet geparsed worden als Double.",
457                     e);
458             throw new ServletException(e);
459         }
