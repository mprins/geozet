8   /**
9    * wrapper om de ResourceBundle met applicatie testen.
10   * 
11   * @author prinsmc@minlnv.nl
12   * @since 1.6
13   */
14  public final class CoreResources {
15      /** log4j logger. */
16      private static final Logger LOGGER = Logger.getLogger(CoreResources.class);
17      /** resources. */
18      private ResourceBundle resBundle;
19  
20      /**
21       * Instantiates a new core resources.
22       * 
23       * @param resBundleName
24       *            the res bundle name
25       */
26      public CoreResources(final String resBundleName) {
27          try {
28              this.resBundle = ResourceBundle
29                      .getBundle(
30                              resBundleName,
31                              ResourceBundle.Control
32                                      .getControl(ResourceBundle.Control.FORMAT_PROPERTIES));
33          } catch (final NullPointerException e) {
34              LOGGER.fatal("baseName or control is null.", e);
35          } catch (final MissingResourceException e) {
36              LOGGER.fatal("De resourcebundel (" + resBundleName
37                      + ".properties) is niet gevonden", e);
38          } catch (final IllegalArgumentException e) {
39              // the given control doesn't perform properly (e.g.,
40              // control.getCandidateLocales returns null.) Note that validation
41              // of control is performed as needed.
42              LOGGER.fatal(
43                      "ResourceBundle.Control heeft geen locale kunnen achterhalen.",
44                      e);
45          }
46      }
47  
48      /**
49       * Geeft de string voor de sleutel, de string kan leeg zijn in het geval van
50       * ontbreken van de key.
51       * 
52       * @param key
53       *            de sleutel
54       * @return de string
55       */
56      public String getString(final String key) {
57          String s = "";
58          try {
59              s = this.resBundle.getString(key);
60          } catch (final MissingResourceException e) {
61              LOGGER.error("Er is geen object gevonden voor de gevraagde key ("
62                      + key + ").", e);
63          } catch (final NullPointerException e) {
64              LOGGER.error("Sleutel (" + key + ") is null.", e);
65          } catch (final ClassCastException e) {
66              LOGGER.error("Het Object voor de gevraagde sleutel  (" + key
67                      + ") is geen String.", e);
68          }
69          return s;
70      }
