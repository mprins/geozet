11  /**
12   * Klasse HttpWrappedRequest wordt gebruikt om vanuit de ene servlet informatie
13   * door te geven aan de volgende via de
14   * {@link javax.servlet.http.HttpServletRequest} . Via de constructor kunnen
15   * extra of aangepaste parameters worden meegegeven.
16   * 
17   * @author strampel@atlis.nl
18   * @since 1.5
19   * @since Servlet API 2.5
20   */
21  public class HttpWrappedRequest extends HttpServletRequestWrapper {
22  
23      /** map met de modifiable request parameters. */
24      private final Map<String, String[]> modifiableParameters;
25  
26      /** map met alle request parameters. */
27      private Map<String, String[]> allParameters = null;
28  
29      /**
30       * Maakt een nieuwe request wrapper waarin additionele request parameters
31       * worden samengevoegd met de reeds bestaande.
32       * 
33       * @param request
34       *            oorspronkelijke request object
35       * @param additionalParams
36       *            additionele request params
37       */
38      public HttpWrappedRequest(final HttpServletRequest request,
39              final Map<String, String[]> additionalParams) {
40          super(request);
41          this.modifiableParameters = new TreeMap<String, String[]>();
42          this.modifiableParameters.putAll(additionalParams);
43      }
44  
45      /*
46       * (non-Javadoc)
47       * 
48       * Geeft de gevraagde parameter of {@code null} als de parameter onbekend
49       * is.
50       * 
51       * @param name the name
52       * 
53       * @return the parameter
54       * 
55       * @see javax.servlet.ServletRequestWrapper#getParameter(java.lang.String)
56       */
57      @Override
58      public String getParameter(final String name) {
59          final String[] strings = this.getParameterMap().get(name);
60          if (strings != null) { return strings[0]; }
61          return null;
62      }
63  
64      /*
65       * (non-Javadoc)
66       * 
67       * @see javax.servlet.ServletRequestWrapper#getParameterMap()
68       */
69      @SuppressWarnings("unchecked")
70      /* servlet api is nog niet generic */
71      @Override
72      public Map<String, String[]> getParameterMap() {
73          if (this.allParameters == null) {
74              this.allParameters = new TreeMap<String, String[]>();
75              this.allParameters.putAll(super.getParameterMap());
76              this.allParameters.putAll(this.modifiableParameters);
77          }
78          return Collections.unmodifiableMap(this.allParameters);
79      }

83       * 
84       * @see javax.servlet.ServletRequestWrapper#getParameterNames()
85       */
86      @Override
