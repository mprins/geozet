104     /*
105      * (non-Javadoc)
106      * 
107      * @see
108      * javax.servlet.http.HttpServlet#service(javax.servlet.http.HttpServletRequest
109      * , javax.servlet.http.HttpServletResponse)
110      */
111     @Override
112     protected void service(HttpServletRequest request,
113             HttpServletResponse response) throws ServletException, IOException {
114 
115         String zoekAdres = request.getParameter(REQ_PARAM_ADRES.code);
116         if ((zoekAdres.length() < 1)
117                 || (zoekAdres.equalsIgnoreCase(this._RESOURCES
118                         .getString("KEY_INTRO_VOORINGEVULD")))) {
119             // als er geen tekst is ingevuld (of de tekst is hetzelfde als die
120             // in de resource bundle)
121             this.renderHTMLResults(request, response, null);
122             response.flushBuffer();
123             return;
124         }
125         // opknippen in de bekende stukjes voor "aansluit specs ingang"
126         if (zoekAdres.startsWith("/")) {
127             // lopende / eraf knippen
128             zoekAdres = zoekAdres.substring(1);
129         }
130         final String params[] = zoekAdres.split("/");
131         LOGGER.debug("gevraagde params: " + Arrays.toString(params));
132 
133         // hashtable met zoektermen en evt. straal
134         String straal = null;
135         final HashMap<String, String> paramTable = new HashMap<String, String>();
136         if (params.length > 1) {
137             for (int i = 0; i < params.length; i = i + 2) {
138                 paramTable.put(params[i].toLowerCase(), params[i + 1]);
139                 LOGGER.debug("toevoegen: " + params[i] + ":" + params[i + 1]);
140             }
141             // proberen straal uit request te halen
142             if (paramTable.containsKey(REQ_PARAM_STRAAL.code)) {
143                 straal = paramTable.remove(REQ_PARAM_STRAAL.code);
144             }
145 
146             final StringBuilder sb = new StringBuilder();
147             for (final String key : StringConstants.urlKeys()) {
148                 if (paramTable.containsKey(key)) {
149                     sb.append(paramTable.get(key)).append(" ");
150                 }
151             }
152             zoekAdres = sb.substring(0, sb.length() - 1).toString();
153         }
154 
155         final Map<String, String> openLSParams = new TreeMap<String, String>();
156 
157         openLSParams.put(OPENLS_REQ_PARAM_REQUEST.code,
158                 OPENLS_REQ_VALUE_GEOCODE.code);
159 
160         openLSParams.put(OPENLS_REQ_PARAM_SEARCH.code, zoekAdres);
161 
162         final GeocodeResponse gcr;
163         if (!paramTable.isEmpty()) {
164             // voor als we kunnen POSTen; vooralsnog werkt dat alleen met de
165             // complexe urls van de aansluit specs
166             gcr = this.openLSClient.doPostOpenLSRequest(this.openLSServerUrl,
167                     this.createGeocodeRequest(paramTable));
168         } else {
169             gcr = this.openLSClient.doGetOpenLSRequest(this.openLSServerUrl,
170                     openLSParams);
171         }
172         final List<OpenLSClientAddress> addrl = OpenLSClientUtil
173                 .getOpenLSClientAddressList(gcr);
174 
175         if (addrl.size() == 1) {
176             // er is maar 1 adres match gevonden
177             final OpenLSClientAddress addr = addrl.get(0);
178             // request parameters voor vervolg opbouwen
179             final Map<String, String[]> extraParams = new HashMap<String, String[]>();
180 
181             extraParams.put(REQ_PARAM_XCOORD.code,
182                     new String[] { this.formatCoord(addr.getxCoord()) });
183             extraParams.put(REQ_PARAM_YCOORD.code,
184                     new String[] { this.formatCoord(addr.getyCoord()) });
185             extraParams
186                     .put(REQ_PARAM_STRAAL.code,
187                             new String[] { (null == straal) ? OPENLS_ZOOMSCALE_STANDAARD
188                                     .toString() : straal });
189             extraParams.put(REQ_PARAM_GEVONDEN.code,
190                     new String[] { addr.getAddressString() });
191 
192             // TODO doorgeven filter
193             // extraParams.put(FILTER_CATEGORIE_NAAM.code,
194             // DataCategorieen.keysAsArray());
195 
196             // doorgeven aan bekendmakingen servlet voor zoekactie
197             final HttpServletRequest wrappedRequest = new HttpWrappedRequest(
198                     request, extraParams);
199             final RequestDispatcher rd = this.getServletContext()
200                     .getRequestDispatcher("/" + this._BEKENDMAKINGEN);
201             rd.forward(wrappedRequest, response);
202         } else {
203             // er zijn 0 of meer dan 1 adressen gevonden
204             this.renderHTMLResults(request, response, addrl);
205             response.flushBuffer();
206         }
207     }

209     /**
210      * zorgt ervoor dat eventuele doubles als integer worden gerenderd. Als het
211      * niet lukt wordt de oorspronkelijke waarde teruggeven.
212      * 
213      * @param coord
214      *            een coordinaat waarde
215      * @return de waarde in integer formaat
216      */
217     private String formatCoord(String coord) {
218         /** coordinaten formatter. */
219         final DecimalFormat fmt = new DecimalFormat("###");
220         try {
221             // formatting als int
222             return "" + (fmt.parse(coord).intValue());
223         } catch (final ParseException e) {
224             LOGGER.warn("Fout tijden parsen van cooridnaat: " + coord, e);
225             return coord;
226         } catch (final NullPointerException e) {
227             LOGGER.warn("Fout tijden parsen van cooridnaat: " + coord, e);
228             return coord;
229         }
230     }

232     /**
233      * maakt een geocode request om te posten naar de service.
234      * 
235      * @param data
236      *            Map met zoekdata uit de url
237      * @return zoekobject
238      */
239     private GeocodeRequest createGeocodeRequest(Map<String, String> data) {
240         final GeocodeRequest gcr = new GeocodeRequest();
241         final Address adres = new Address();
242         if (data.containsKey(URL_KEY_PROVINCIE.code)) {
243             final Place p = new Place();
244             p.setPlace(data.get(URL_KEY_PROVINCIE.code));
245             p.setType(OpenLSConstants.PLACE_TYPE_COUNTRYSUBDIVISION);
246             adres.addPlace(p);
247         }
248         if (data.containsKey(URL_KEY_GEMEENTE.code)) {
249             final Place p = new Place();
250             p.setPlace(data.get(URL_KEY_GEMEENTE.code));
251             p.setType(OpenLSConstants.PLACE_TYPE_MUNICIPALITY);
252             adres.addPlace(p);
253         }
254         if (data.containsKey(URL_KEY_PLAATS.code)) {
255             final Place p = new Place();
256             p.setPlace(data.get(URL_KEY_PLAATS.code));
257             p.setType(OpenLSConstants.PLACE_TYPE_MUNICIPALITYSUBDIVISION);
258             adres.addPlace(p);
259         }
260         if (data.containsKey(URL_KEY_WIJK.code)) {
261             final Place p = new Place();
262             p.setPlace(data.get(URL_KEY_WIJK.code));
263             p.setType(OpenLSConstants.PLACE_TYPE_MUNICIPALITYSUBDIVISION);
264             adres.addPlace(p);
265         }
266         if (data.containsKey(URL_KEY_BUURT.code)) {
267             final Place p = new Place();
268             p.setPlace(data.get(URL_KEY_BUURT.code));
269             p.setType(OpenLSConstants.PLACE_TYPE_MUNICIPALITYSUBDIVISION);
270             adres.addPlace(p);
271         }
272         if (data.containsKey(URL_KEY_POSTCODE.code)) {
273             final PostalCode p = (new PostalCode());
274             p.setPostalCode(data.get(URL_KEY_POSTCODE.code));
275             adres.setPostalCode(p);
276         }
277         if (data.containsKey(URL_KEY_STRAAT.code)) {
278             final StreetAddress sa = new StreetAddress();
279             final Street s = new Street();
280             s.setStreet(data.get(URL_KEY_STRAAT.code));
281             sa.setStreet(s);
282             if (data.containsKey(URL_KEY_HUISNUMMER.code)) {
283                 final Building b = new Building();
284                 b.setNumber(data.get(URL_KEY_HUISNUMMER.code));
285                 sa.setBuilding(b);
286             }
287             adres.setStreetAddress(sa);
288         }
289         gcr.addAddress(adres);
290         return gcr;
291     }

293     /**
294      * Renderen van de features in html formaat, wordt alleen aangeroepen in het
295      * geval de lijst met adressen meer of minder dan 1 object bevat..
296      * 
297      * @param request
298      *            servlet request
299      * @param response
300      *            servlet response
301      * @param addrl
302      *            Een lijst met adressen, de lengte van de lijst is (voor deze
303      *            applicatie) nul of groter dan een, in het geval null dan is er
304      *            geen zoekter ingevuld geweest.
305      * @throws IOException
306      *             als er een schrijffout optreedt
307      * @throws ServletException
308      *             the servlet exception
309      */
310     private void renderHTMLResults(HttpServletRequest request,
311             HttpServletResponse response, List<OpenLSClientAddress> addrl)
312             throws IOException, ServletException {
313         // response headers instellen
314         response.setContentType("text/html; charset=UTF-8");
315 
316         // header inhaken
317         final RequestDispatcher header = this.getServletContext()
318                 .getRequestDispatcher("/WEB-INF/jsp/locatielijst_begin.jsp");
319         if (header != null) {
320             header.include(request, response);
321         }
322 
323         final StringBuilder sbResult = new StringBuilder();
324         StringBuilder sb;
325 
326         if (addrl == null) {
327             // niks ingevuld geweest om naar te zoeken
328             sbResult.append("<p class=\"geozetError\">")
329                     .append(this._RESOURCES.getString("KEY_FOUT_GEEN_INPUT"))
330                     .append("</p>\n");
331         } else if (addrl.isEmpty()) {
332             // niks gevonden
333             sbResult.append("<p class=\"geozetError\">")
334                     .append(this._RESOURCES
335                             .getString("KEY_FOUT_LOCATIE_NIET_GEVONDEN"))
336                     .append("</p>\n");
337         } else {
338             // meer dan 1 gevonden
339             sbResult.append("<p class=\"geozetError\">")
340                     .append(this._RESOURCES
341                             .getString("KEY_FOUT_MEERDERELOCATIES_GEVONDEN"))
342                     .append("</p>\n");
343 
344             sbResult.append(this._RESOURCES.getString("KEY_FOUT_BEDOELDE_U"));
345 
346             sbResult.append("<ul class=\"geozetRecommends\">\n");
347 
348             // door lijst adressen heen gaan
349             for (final OpenLSClientAddress addr : addrl) {
350                 // samenstellen URL
351                 sb = new StringBuilder();
352                 sb.append(this._BEKENDMAKINGEN).append("?");
353 
354                 sb.append(REQ_PARAM_XCOORD).append("=")
355                         .append(this.formatCoord(addr.getxCoord()))
356                         .append("&amp;");
357                 sb.append(REQ_PARAM_YCOORD).append("=")
358                         .append(this.formatCoord(addr.getyCoord()))
359                         .append("&amp;");
360                 // straal
361                 String straal;
362                 final String reqstraal = request
363                         .getParameter(REQ_PARAM_STRAAL.code);
364                 LOGGER.debug("request straal is: " + reqstraal);
365                 if (reqstraal == null) {
366                     // als de straal niet opgegeven is proberen op basis van de
367                     // OLS hit te bepalen wat de staal zou moeten zijn
368                     if (addr.getAddressString().contains(
369                             OpenLSClientAddress.APPEND_PLAATS)) {
370                         // plaats
371                         straal = OPENLS_ZOOMSCALE_PLAATS.toString();
372                     } else if (addr.getAddressString().contains(
373                             OpenLSClientAddress.APPEND_GEMEENTE)) {
374                         // gemeente
375                         straal = OPENLS_ZOOMSCALE_GEMEENTE.toString();
376                     } else if (addr.getAddressString().contains(
377                             OpenLSClientAddress.APPEND_PROVINCIE)) {
378                         // provincie
379                         straal = OPENLS_ZOOMSCALE_PROVINCIE.toString();
380                     } else {
381                         // default
382                         straal = OPENLS_ZOOMSCALE_STANDAARD.toString();
383                     }
384                 } else {
385                     straal = reqstraal;
386                 }
387                 sb.append(REQ_PARAM_STRAAL).append("=").append(straal)
388                         .append("&amp;");
389 
390                 // sb.append(REQ_PARAM_ADRES).append("=")
391                 // .append(addr.getAddressString()).append("&amp;");
392                 sb.append(REQ_PARAM_ADRES).append("=")
393                         .append(request.getParameter(REQ_PARAM_ADRES.code))
394                         .append("&amp;");
395 
396                 sb.append(REQ_PARAM_GEVONDEN).append("=")
397                         .append(addr.getAddressString());
398                 // coreonly alleen als die aanwezig is
399                 if (null != request.getParameter(REQ_PARAM_COREONLY.code)) {
400                     sb.append("&amp;")
401                             .append(REQ_PARAM_COREONLY)
402                             .append("=")
403                             .append(request
404                                     .getParameter(REQ_PARAM_COREONLY.code));
405                 }
406                 // de URL is uiteindelijke HTML/hyperlink output
407                 sbResult.append("<li><a href=\"").append(sb.toString())
408                         .append("\">");
409                 sbResult.append(addr.getAddressString()).append("</a></li>\n");
410             }
411             sbResult.append(" </ul>");
412         }
413         final PrintWriter out = response.getWriter();
414         out.print(sbResult);
415         out.flush();
416         // footer inhaken
417         final RequestDispatcher footer = this.getServletContext()
418                 .getRequestDispatcher("/WEB-INF/jsp/locatielijst_einde.jsp");
419         if (footer != null) {
420             footer.include(request, response);
421         }
422     }
