48      /**
49       * Maakt een vierkante bbox met RD CRS met lengte en hoogte van
50       * {@code 2 x afstand} (dus straal {@code afstand}) om het het
51       * coordinatenpaar {@code xcoord};{@code ycoord}.
52       * 
53       * @param xcoord
54       *            xcoord
55       * @param ycoord
56       *            ycoord
57       * @param afstand
58       *            afstand
59       * @return the bounding box
60       */
61      public static BoundingBox calcRDBBOX(double xcoord, double ycoord,
62              double afstand) {
63          try {
64              return new Envelope2D(CRS.decode("EPSG:28992"), (xcoord - afstand),
65                      (ycoord - afstand), 2 * afstand, 2 * afstand);
66          } catch (final NoSuchAuthorityCodeException e) {
67              LOGGER.fatal("De gevraagde CRS autoriteit is niet gevonden.", e);
68          } catch (final FactoryException e) {
69              LOGGER.fatal(
70                      "Gevraagde GeoTools factory voor CRS is niet gevonden.", e);
71          }
72          return null;
73      }
