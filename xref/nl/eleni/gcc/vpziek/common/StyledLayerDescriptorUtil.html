186     /**
187      * Creates the bw style.
188      * 
189      * @param layernames
190      *            the layernames
191      * @param requestedFontSize
192      *            the requested font size
193      * @param classNames
194      *            the class names
195      * @return the styled layer descriptor
196      * @throws TransformerException
197      *             the transformer exception
198      */
199     public static StyledLayerDescriptor createBWStyle(String[] layernames,
200             int requestedFontSize, List<String> classNames)
201             throws TransformerException {
202 
203         // lijst maken van alle beschikbare marks
204         final String[] marknames = sb.getWellKnownMarkNames();
205         final Mark[] marks = new Mark[marknames.length];
206         for (int i = 0; i < marknames.length; i++) {
207             marks[i] = sb.createMark(marknames[i], Color.BLACK, 1);
208         }
209 
210         int markCounter = -1;
211         int ruleCounter = 0;
212 
213         final Rule[] rules = new Rule[classNames.size()];
214 
215         for (final String actName : classNames) {
216             // loop door actieve klassen, een rule voor iedere klasse maken
217 
218             if (markCounter++ > marks.length) {
219                 // loop markers als we aan het eind zijn
220                 markCounter = 0;
221             }
222 
223             final Graphic g = sb.createGraphic(null, marks[markCounter], null);
224             final Fill fill = sb.createFill();
225             fill.setGraphicFill(g);
226             final PolygonSymbolizer sym = sb.createPolygonSymbolizer(
227                     OUTLINESTROKE, fill);
228 
229             final Rule rule = sb.createRule(new Symbolizer[] { sym });
230             rule.setFilter(filterFactory.equals(
231                     filterFactory.property(FEATURE_ATTR_NAAM_TITEL.code),
232                     filterFactory.literal(actName)));
233             rules[ruleCounter++] = rule;
234         }
235 
236         final FeatureTypeStyle fts = sb
237                 .createFeatureTypeStyle("Feature", rules);
238 
239         final TextSymbolizer txt = sb.createTextSymbolizer(Color.BLACK,
240                 sb.createFont("sans-serif", true, true, requestedFontSize),
241                 FEATURE_ATTR_NAAM_TITEL.code);
242         fts.rules().add(sb.createRule(new Symbolizer[] { txt }));
243         fts.setName(null);
244 
245         final Style style = sb.createStyle();
246         style.setName(SLD_STYLE_NAME);
247         style.featureTypeStyles().add(fts);
248 
249         final StyledLayerDescriptor sld = styleFactory
250                 .createStyledLayerDescriptor();
251 
252         for (final String lyrName : layernames) {
253             // voor iedere laag dezelfde styling
254             final NamedLayer named = styleFactory.createNamedLayer();
255             named.addStyle(style);
256             named.setName(lyrName);
257             sld.addStyledLayer(named);
258         }
259 
260         LOGGER.debug((new SLDTransformer()).transform(sld));
261         return sld;
262     }

373     /**
374      * Create a style.
375      * 
376      * @param layernames
377      *            layernames waarvoor style geldig moet zij
378      * @param requestedFontSize
379      *            font size voor labels
380      * @param classNames
381      *            namen van de klassen
382      * @param colours
383      *            kleuren van de klassen, dit array wordt round-robin gebruikt,
384      *            daardoor mag de lengte kleiner zijn dan het aantal klassen
385      * @return de styled layer descriptor
386      * @throws TransformerException
387      *             treed op als de transformatie van de sld naar een string
388      *             mislukt
389      */
390     public static StyledLayerDescriptor createStyle(String[] layernames,
391             int requestedFontSize, List<String> classNames, Color[] colours)
392             throws TransformerException {
393 
394         final Rule[] rules = new Rule[classNames.size()];
395         int ruleCounter = 0;
396         int colCounter = 0;
397         for (final String actName : classNames) {
398             // loop actieve klassen, een rule voor iedere klassificatie maken
399 
400             final Color col = colours[colCounter++];
401             if (colCounter == colours.length) {
402                 // begin weer vooraan met kleuren als we aan het eind zijn
403                 colCounter = 0;
404             }
405 
406             final PolygonSymbolizer polySymb = sb.createPolygonSymbolizer(
407                     OUTLINESTROKE, sb.createFill(col, FILLOPACITY), null);
408 
409             final Rule rule = sb.createRule(new Symbolizer[] { polySymb });
410             rule.setFilter(
411             /* titel = ziekte */
412             filterFactory.and(filterFactory.equals(
413                     filterFactory.property(FEATURE_ATTR_NAAM_TITEL.code),
414                     filterFactory.literal(actName)),
415             // TODO verfijnen alleen actieve gebieden op basis van einddatum
416             // (FEATURE_ATTR_NAAM_EINDDATUM)
417                     /* einddatum == null */
418                     filterFactory.isNull(filterFactory
419                             .property(FEATURE_ATTR_NAAM_EINDDATUM.code))));
420             rules[ruleCounter++] = rule;
421         }
422 
423         // txt symbolizer
424         final TextSymbolizer txtSymb = sb.createTextSymbolizer(Color.BLACK,
425                 sb.createFont("sans-serif", true, true, requestedFontSize),
426                 FEATURE_ATTR_NAAM_TITEL.code);
427         // AGS doet geen halo..
428         // txtSymb.setHalo(sb.createHalo(Color.WHITE, 2));
429 
430         final FeatureTypeStyle fts = sb
431                 .createFeatureTypeStyle("Feature", rules);
432         fts.rules().add(sb.createRule(new Symbolizer[] { txtSymb }));
433         fts.setName(null);
434 
435         final Style style = sb.createStyle();
436         style.setName(SLD_STYLE_NAME);
437         style.featureTypeStyles().add(fts);
438 
439         final StyledLayerDescriptor sld = styleFactory
440                 .createStyledLayerDescriptor();
441 
442         for (final String lyrName : layernames) {
443             // voor iedere laag dezelfde styling
444             final NamedLayer named = styleFactory.createNamedLayer();
445             named.addStyle(style);
446             named.setName(lyrName);
447             sld.addStyledLayer(named);
448         }
449 
450         LOGGER.debug((new SLDTransformer()).transform(sld));
451         return sld;
452     }

454     /**
455      * Creates the color range.
456      * 
457      * @param classes
458      *            the classes
459      * @return the color[]
460      * @todo evt. parametriseren met props file
461      */
462     static Color[] createColorRange(int classes) {
463         final Color[] colors = new Color[] { Color.RED, Color.YELLOW,
464                 Color.ORANGE, Color.CYAN, Color.BLUE, Color.PINK };
465         if (classes > colors.length) {
466             final Color[] newColors = Arrays.copyOf(colors, classes);
467             for (int i = colors.length; i < newColors.length; i++) {
468                 newColors[i] = colors[(i % colors.length)];
469             }
470             LOGGER.debug(Arrays.deepToString(newColors));
471             return newColors;
472         } else {
473             return Arrays.copyOf(colors, classes);
474         }
475     }

477     /**
478      * Creates de grey scale range, ingebouwde lengte is 11.
479      * 
480      * @param classes
481      *            the classes
482      * @return the color[]
483      * @todo evt. parametriseren met props file
484      */
485     static Color[] createGreyScaleRange(int classes) {
486         final Color[] colors = new Color[classes];
487         final int colStep = 20;
488         int colStart = 240;
489         for (int i = 0; i < classes; i++) {
490             colors[i] = new Color(colStart, colStart, colStart);
491             colStart = colStart - colStep;
492             if (colStart < colStep) {
493                 colStart = 240;
494             }
495         }
496         LOGGER.debug(Arrays.deepToString(colors));
497         return colors;
498     }
