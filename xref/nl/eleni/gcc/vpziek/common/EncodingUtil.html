13  public class EncodingUtil {

17      private EncodingUtil() {

20      /**
21       * Decodes the passed UTF-8 String using an algorithm that's compatible with
22       * JavaScript's <code>decodeURIComponent</code> function. Returns
23       * <code>null</code> if the String is <code>null</code>.
24       * 
25       * @param s
26       *            The UTF-8 encoded String to be decoded
27       * @return the decoded String
28       */
29      public static String decodeURIComponent(String s) {
30          if (s == null) {
31              return null;
32          }
33          String result = null;
34          try {
35              result = URLDecoder.decode(s, "UTF-8");
36          } catch (final UnsupportedEncodingException e) {
37              // This exception should never occur.
38              result = s;
39          }
40          return result;
41      }
42  
43      /**
44       * Encodes the passed String as UTF-8 using an algorithm that's compatible
45       * with JavaScript's <code>encodeURIComponent</code> function. Returns
46       * <code>null</code> if the String is <code>null</code>.
47       * 
48       * @param s
49       *            The String to be encoded
50       * @return the encoded String
51       */
52      public static String encodeURIComponent(String s) {
53          String result = null;
54          try {
55              result = URLEncoder.encode(s, "UTF-8").replaceAll("\\+", "%20")
56                      .replaceAll("\\%21", "!").replaceAll("\\%27", "'")
57                      .replaceAll("\\%28", "(").replaceAll("\\%29", ")")
58                      .replaceAll("\\%7E", "~");
59          } catch (final UnsupportedEncodingException e) {
60              // This exception should never occur.
61              result = s;
62          }
63          return result;
64      }
