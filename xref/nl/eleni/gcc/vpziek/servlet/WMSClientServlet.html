182     protected void service(HttpServletRequest request,
183             HttpServletResponse response) throws ServletException, IOException {
184         final double[] dXcoordYCoordStraal = this.parseLocation(request);
185         final double xcoord = dXcoordYCoordStraal[0];
186         final double ycoord = dXcoordYCoordStraal[1];
187         final double straal = dXcoordYCoordStraal[2];
188 
189         final ActieveZiektenBean actieveZiektenBean = (ActieveZiektenBean) this
190                 .getServletContext().getAttribute("actieveziekten");
191         List<String> actieveZiektenamen = actieveZiektenBean.getElements();
192 
193         final String[] fString = request
194                 .getParameterValues(REQ_PARAM_FILTER.code);
195         final String filterUsed = request
196                 .getParameter(REQ_PARAM_EXPLICITUSEFILTER.code);
197 
198         if ((null == fString) && (null != filterUsed)
199                 && filterUsed.equalsIgnoreCase("true")) {
200             LOGGER.debug("Er is expres gezocht met een leeg filter");
201             // TODO wat nu ?? geen kaart?!
202             // this.renderHTMLResults(request, response, null);
203             // return;
204             // toch een kaart
205             actieveZiektenamen = Collections.emptyList();
206         } else if (null != fString) {
207             LOGGER.debug("Er wordt gezocht met een filter");
208             actieveZiektenamen = Arrays.asList(fString);
209         } else {
210             LOGGER.debug("Er wordt gezocht zonder een filter");
211         }
212 
213         // parse kleuren schema uit verzoek
214         ColourStyleEnum colStyle = ColourStyleEnum.KL;
215         final String cScheme = request.getParameter(REQ_PARAM_COLORSCHEME.code);
216         LOGGER.debug("request params; col schema: " + cScheme);
217         try {
218             if (cScheme != null) {
219                 colStyle = ColourStyleEnum.valueOf(cScheme);
220             }
221         } catch (final IllegalArgumentException e) {
222             LOGGER.warn("Er is geen (of een onbekende) stijl meegegeven voor de kaart. De default wordt gebruikt.");
223         }
224 
225         // parse font grootte
226         int requestedFontSize = DEFAULT_FONT_SIZE.intValue();
227         try {
228             requestedFontSize = Integer.parseInt(request
229                     .getParameter(REQ_PARAM_FONTSIZE.code));
230         } catch (final NullPointerException e) {
231             LOGGER.warn("De parameter " + REQ_PARAM_FONTSIZE

241         try {
242             final File img = this.getMap(bbox, requestedFontSize, colStyle,

253     /**
254      * kaart ophalen.
255      * 
256      * @param bbox
257      *            the bbox
258      * @param requestedFontSize
259      *            the requested font size
260      * @param style
261      *            the style
262      * @param layers
263      *            the layers
264      * @param actieveZiektenamen
265      *            the actieve ziektenamen
266      * @return the map
267      * @throws ServiceException
268      *             the service exception
269      * @throws IOException
270      *             Signals that an I/O exception has occurred.
271      */
272     private File getMap(BoundingBox bbox, int requestedFontSize,
273             ColourStyleEnum style, String[] layers,
274             List<String> actieveZiektenamen) throws ServiceException,
275             IOException {
276 
277         Color drawCol = Color.MAGENTA;
278         BufferedImage image = new BufferedImage(MAP_DIMENSION, MAP_DIMENSION,
279                 BufferedImage.TYPE_INT_ARGB);
280         if (!actieveZiektenamen.isEmpty()) {
281             // wms request doen
282             final GetMapRequest map = this.fgWMS.createGetMapRequest();
283             map.addLayer(layers[0], StyledLayerDescriptorUtil.SLD_STYLE_NAME);
284             map.addLayer(layers[1], StyledLayerDescriptorUtil.SLD_STYLE_NAME);
285             map.setFormat("image/png32");
286             map.setDimensions(MAP_DIMENSION, MAP_DIMENSION);
287             map.setTransparent(true);
288             map.setSRS("EPSG:28992");
289             map.setBBox(bbox);
290             map.setExceptions("application/vnd.ogc.se_inimage");
291             try {
292                 StyledLayerDescriptor sld;
293                 switch (style) {
294                 case MON:
295                     sld = StyledLayerDescriptorUtil.createDistinctColourStyle(
296                             layers, requestedFontSize, actieveZiektenamen);
297                     break;
298                 case GR:
299                     sld = StyledLayerDescriptorUtil.createGreyscaleStyle(
300                             layers, requestedFontSize, actieveZiektenamen);
301                     drawCol = Color.BLACK;
302                     break;
303                 case ZW:
304                     sld = StyledLayerDescriptorUtil.createBWStyle(layers,
305                             requestedFontSize, actieveZiektenamen);
306                     drawCol = Color.BLACK;
307                     break;
308                 case KL:
309                     // doorvallen, KL is de default
310                 default:
311                     sld = StyledLayerDescriptorUtil.createFullColourStyle(
312                             layers, requestedFontSize, actieveZiektenamen);
313                 }
314                 final SLDTransformer trans = new SLDTransformer();
315                 /*
316                  * vanwege een bug in arcgis server de xml document prolog
317                  * verwijderen van de sld
318                  */
319                 trans.setOmitXMLDeclaration(true);
320                 final String sldBody = trans.transform(sld);
321                 LOGGER.debug("De volgende SLD_BODY wordt verstuurd: " + sldBody);
322                 map.setProperty(GetMapRequest.SLD_BODY,
323                         encodeURIComponent(sldBody));
324             } catch (final TransformerException e) {
325                 LOGGER.error(
326                         "Er is een fout opgetreden tijdens de transformatie van het gegenereerde SLD document.",
327                         e);
328             }
329             map.setBGColour("0xffffff");
330             LOGGER.debug("Voorgrond WMS url is: " + map.getFinalURL());
331 
332             // thema/voorgrond ophalen
333             final GetMapResponse response = this.fgWMS.issueRequest(map);
334             image = ImageIO.read(response.getInputStream());
335             if (LOGGER.isDebugEnabled()) {
336                 // voorgrond plaatje bewaren in debug modus
337                 final File temp = File.createTempFile(
338                         "fgwms",
339                         ".png",
340                         new File(this.getServletContext().getRealPath(
341                                 MAP_CACHE_DIR)));
342                 temp.deleteOnExit();
343                 ImageIO.write(image, "png", temp);
344             }
345 
346         } else {
347             // geen wms request doen
348         }
349 
350         final BufferedImage image2 = this.getBackGround(bbox);
351         final BufferedImage composite = new BufferedImage(MAP_DIMENSION,
352                 MAP_DIMENSION, BufferedImage.TYPE_INT_ARGB);
353         final Graphics g = composite.getGraphics();
354         g.drawImage(image2, 0, 0, null);
355         g.drawImage(image, 0, 0, null);
356 
357         // zoeklocatie intekenen met halo
358         final int width = 4;
359         final int[] px = { MAP_DIMENSION_MIDDLE - width,
360                 MAP_DIMENSION_MIDDLE + width, MAP_DIMENSION_MIDDLE };
361         final int[] py = { MAP_DIMENSION_MIDDLE + width,
362                 MAP_DIMENSION_MIDDLE + width, MAP_DIMENSION_MIDDLE - width };
363         final int offset = 2;
364         final int[] pxh = { px[0] - offset, px[1] + offset, px[2] };
365         final int[] pyh = { py[0] + offset, py[1] + offset, py[2] - offset };
366 
367         g.setFont(new Font(Font.SANS_SERIF, Font.BOLD, requestedFontSize));
368         // witte halo voor text/icoon
369         g.setColor(Color.WHITE);
370         g.drawString("zoeklocatie", MAP_DIMENSION_MIDDLE + 5,
371                 MAP_DIMENSION_MIDDLE + 5);
372         g.drawString("zoeklocatie", MAP_DIMENSION_MIDDLE + 5,
373                 MAP_DIMENSION_MIDDLE + 7);
374         g.drawString("zoeklocatie", MAP_DIMENSION_MIDDLE + 7,
375                 MAP_DIMENSION_MIDDLE + 7);
376         g.drawString("zoeklocatie", MAP_DIMENSION_MIDDLE + 7,
377                 MAP_DIMENSION_MIDDLE + 5);
378         g.fillPolygon(pxh, pyh, pxh.length);
379 
380         // text/ikoon
381         g.setColor(drawCol);
382         g.fillPolygon(px, py, px.length);
383         g.drawString("zoeklocatie", MAP_DIMENSION_MIDDLE + 6,
384                 MAP_DIMENSION_MIDDLE + 6);
385         // opslaan van plaatje zodat de browser het op kan halen
386         final File temp3 = File.createTempFile("wmscombined", ".png", new File(
387                 this.getServletContext().getRealPath(MAP_CACHE_DIR)));
388         temp3.deleteOnExit();
389         ImageIO.write(composite, "png", temp3);
390 
391         return temp3;
392

395     /**
396      * Achtergrondkaart ophalen.
397      * 
398      * @param bbox
399      *            the bbox
400      * @return background/basemap image
401      * @throws ServiceException
402      *             the service exception
403      * @throws IOException
404      *             Signals that an I/O exception has occurred.
405      */
406     private BufferedImage getBackGround(BoundingBox bbox)
407             throws ServiceException, IOException {
408 
409         if (this.bgWMSCache.containsKey(bbox)) {
410             // check cache
411             return this.bgWMSCache.get(bbox);
412         }
413 
414         final GetMapRequest map = this.bgWMS.createGetMapRequest();
415         if (this.bgWMSlayers != null) {
416             for (final String lyr : this.bgWMSlayers) {
417                 // per laag toevoegen met de default style
418                 map.addLayer(lyr, "");
419             }
420         } else {
421             // alle lagen toevoegen
422             for (final Layer layer : WMSUtils.getNamedLayers(this.bgWMS
423                     .getCapabilities())) {
424                 map.addLayer(layer);
425             }
426         }
427 
428         map.setFormat("image/png");
429         map.setDimensions(MAP_DIMENSION, MAP_DIMENSION);
430         map.setTransparent(true);
431         map.setBGColour("0xffffff");
432         map.setExceptions("application/vnd.ogc.se_inimage");
433         map.setSRS("EPSG:28992");
434         map.setBBox(bbox);
435 
436         LOGGER.debug("Achtergrond WMS url is: " + map.getFinalURL());
437 
438         final GetMapResponse response = this.bgWMS.issueRequest(map);
439         final BufferedImage image = ImageIO.read(response.getInputStream());
440         this.bgWMSCache.put(bbox, image);
441         return image;
442     }
